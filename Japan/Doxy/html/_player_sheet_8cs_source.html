122 int HPscaleLevel(int level)
123 {
124 //initial HP dice in OD&D are determined by class, but as we are as yet classless, it needs to be
125 //some coefficient of STR and CON - so just invent it.
126 //NOTE: diverging from D&D giving mages +0, clerics +1, and fighters +2
127
128 int hp = 0;
129 float scale = (STR + CON) * 0.01f;
130 //18 + 18 = 36 which would be the max possible, so scale < .12 magic, < .24 cleric < .36 fighter
131 hp = Random.Range(1, 7);
132
133 if (scale < 0.12f)
134 hp += 0;
135 else if (scale < 0.24f)
136 hp += 1;
137 else if (scale < 0.36)
138 hp += 2;
139
140 return hp * level; //TODO: this might need rethunk as characters level up
